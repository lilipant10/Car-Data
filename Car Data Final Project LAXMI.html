from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score, mean_squared_error, mean_absolute_error
from sklearn.model_selection import train_test_split
import numpy as np

# Assuming 'milage' as the target variable
# Exclude 'milage' and non-numeric columns from features
numeric_columns = df.select_dtypes(include=['number']).columns.tolist()
X = df[numeric_columns].drop(columns=['milage'])
y = df['milage']

# Train-test split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Define evaluation function
def evaluate_model(model, X_test, y_test):
    y_pred = model.predict(X_test)
    r2 = r2_score(y_test, y_pred)
    adj_r2 = 1 - (1-r2) * (len(y_test) - 1) / (len(y_test) - X_test.shape[1] - 1)
    mape = np.mean(np.abs((y_test - y_pred) / y_test)) * 100
    rmse = np.sqrt(mean_squared_error(y_test, y_pred))
    mae = mean_absolute_error(y_test, y_pred)
    return r2, adj_r2, mape, rmse, mae

# Linear Regression
lr = LinearRegression()
lr.fit(X_train, y_train)
lr_r2, lr_adj_r2, lr_mape, lr_rmse, lr_mae = evaluate_model(lr, X_test, y_test)

# Print evaluation metrics
print(f"Linear Regression: \n"
      f"R2 = {lr_r2:.2f}, \n"
      f"Adjusted R2 = {lr_adj_r2:.2f}, \n"
      f"MAPE = {lr_mape:.2f}%, \n"
      f"RMSE = {lr_rmse:.2f}, \n"
      f"MAE = {lr_mae:.2f}")
